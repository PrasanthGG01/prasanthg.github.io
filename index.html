-- Patient admission analysis
SELECT patient_id, 
    admission_type,
    COUNT(*) AS total_admissions,
    AVG(length_of_stay) AS avg_stay_days
FROM patient_admissions
WHERE admission_date >= '2023-01-01'
GROUP BY patient_id, admission_type
ORDER BY total_admissions DESC;
